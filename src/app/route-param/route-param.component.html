  • Required Params: /myId
      # Navigation:
    • /myId
      # Accessing:
    • .snapshot.paramMap: ParamMap -{{requiredParamSnapshot}}
    • .paramMap: Observable<ParamMap> - {{requiredParam$ | async}}
  • Optional Params:
      # Navigation:
    • /myId;optId=myOptId
    • [routerLink]="['/routeparam', 'myId', { optId: 'myOptId' }]"
      # Accessing:

    • .paramMap: Observable<ParamMap> {{optionalParam$ | async}}

  • Query param:
      # Navigation:
    • /myId?id=myId
    • [routerLink]="['/routeparam', 'myId']" [queryParams]="{ id: 'myId' }"
    • .navigate(['/routeparam', 'myId'], { queryParams: { id: 'myId' } });
    • Preserve Query Params - queryParamsHandling:
      • [routerLink]="['/home']" queryParamsHandling="preserve"
      # Accessing:
    • .queryParamMap: Observable<ParamMap> {{queryParam$ | async}}
